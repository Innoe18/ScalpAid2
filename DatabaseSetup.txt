DROP DATABASE IF EXISTS scalpaid;
CREATE DATABASE scalpaid;
USE scalpaid;

DROP TABLE IF EXISTS users, sensor_readings;

CREATE TABLE users (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(100),
  email VARCHAR(190) NOT NULL UNIQUE,
  hair_type VARCHAR(50),
  purpose VARCHAR(255),
  password VARCHAR(255) NOT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE sensor_readings (
  id INT AUTO_INCREMENT PRIMARY KEY,
  device_id VARCHAR(64) NOT NULL,
  moisture_cipher TEXT NOT NULL,
  temperature_cipher TEXT NOT NULL,
  moisture_unit VARCHAR(16) DEFAULT '%',
  temperature_unit VARCHAR(16) DEFAULT 'C',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  KEY idx_device_time (device_id, created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
